async function loadProjects() { const items = document.querySelectorAll('.project-item'); for (const item of items) { const yt = item.getAttribute('data-youtube'); const img = item.getAttribute('data-image'); const descCustom = item.getAttribute('data-description'); const media = item.querySelector('.project-media'); const descEl = item.querySelector('.description'); // DEFAULT media.innerHTML = ``; descEl.innerText = descCustom || ""; // Nếu có ảnh riêng if (img) { media.innerHTML = ``; } // Nếu có youtube → override if (yt) { try { const res = await fetch(`https://noembed.com/embed?url=${yt}`); const data = await res.json(); // Thumbnail media.innerHTML = ``; // Title + Description let text = data.title || ""; if (data.author_name) { text += "\n" + data.author_name; } descEl.innerText = text; } catch (e) { console.error(e); } } } }
